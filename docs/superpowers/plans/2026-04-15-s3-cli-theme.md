# S3 — Visual CLI Theme Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Give the loomflo CLI a coherent pastel-green (Mint) visual identity by centralising all terminal output through a single theme module, with proper respect for `NO_COLOR`, non-TTY, and `--json` modes.

**Architecture:** A small `theme.ts` module in the CLI package exposes semantic colour functions (`accent`/`muted`/`dim`/`warn`/`err`), glyphs, and layout helpers (`heading`/`kv`/`line`/`table`/`spinner`) backed by `chalk@5`, `ora@8`, and `cli-table3`. A shared `output.ts` provides `withJsonSupport()` so every command cleanly toggles between themed output and machine-readable JSON. Existing commands are migrated file-by-file to route all output through these helpers; a lint rule prevents regression.

**Tech Stack:** TypeScript 5.x, Node 20+, `chalk@5` (truecolour), `ora@8` (spinners), `cli-table3` (tables), `vitest` (tests), `tsup` (bundle), `eslint` (regression guard).

**Spec:** `docs/superpowers/specs/2026-04-15-s3-cli-theme.md`

```ts
import Table from "cli-table3";
import chalk from "chalk";
import ora, { type Ora } from "ora";

import { palette, type Tone } from "./palette.js";

const noColor = (s: string): string => s;

const toneFn = (tone: Tone): ((s: string) => string) => {
  if (chalk.level === 0) return noColor;
  const [r, g, b] = palette[tone];
  return (s: string): string => chalk.rgb(r, g, b)(s);
};

export const glyph = {
  check: "✓",
  cross: "✗",
  arrow: "→",
  warn: "⚠",
  dot: "●",
} as const;

export interface Column<T> {
  header: string;
  get: (row: T) => string;
  /** Optional alignment. Defaults to "left". */
  align?: "left" | "right" | "center";
}

/**
 * Wraps cli-table3 to produce a consistent table with ASCII borders when
 * colour is disabled and Unicode borders otherwise.
 */
function renderTable<T>(headers: string[], rows: T[], columns: Column<T>[]): string {
  const useAscii = chalk.level === 0;
  const chars = useAscii
    ? {
        top: "-",
        "top-mid": "+",
        "top-left": "+",
        "top-right": "+",
        bottom: "-",
        "bottom-mid": "+",
        "bottom-left": "+",
        "bottom-right": "+",
        left: "|",
        "left-mid": "+",
        mid: "-",
        "mid-mid": "+",
        right: "|",
        "right-mid": "+",
        middle: "|",
      }
    : undefined;

  const table = new Table({
    head: headers,
    chars,
    style: { head: useAscii ? [] : ["bold"], border: useAscii ? [] : [] },
  });

  for (const row of rows) {
    table.push(columns.map((c) => c.get(row)));
  }

  return table.toString();
}

function headingFn(text: string): string {
  const colored = toneFn("accent")(text);
  const underline = "-".repeat(text.length);
  return `${colored}\n${underline}`;
}

function kvFn(key: string, value: string, keyWidth = 9): string {
  const padded = key.padEnd(keyWidth, " ");
  return `  ${toneFn("muted")(padded)}  ${value}`;
}

function lineFn(g: string, tone: Tone, text: string, meta?: string): string {
  const head = `${toneFn(tone)(g)} ${text}`;
  if (meta === undefined) return head;
  return `${head}  ${toneFn("dim")(`(${meta})`)}`;
}

function spinnerFn(text: string): Ora {
  return ora({
    text,
    color: "green",
    spinner: "dots",
    stream: process.stderr,
  });
}

export const theme = {
  accent: toneFn("accent"),
  muted:  toneFn("muted"),
  dim:    toneFn("dim"),
  warn:   toneFn("warn"),
  err:    toneFn("err"),
  glyph,
  heading: headingFn,
  kv: kvFn,
  line: lineFn,
  spinner: spinnerFn,
  table: renderTable,
};

export type Theme = typeof theme;
```

```ts
import type { Command } from "commander";

import { theme } from "./theme/index.js";

export interface WithJsonOption {
  json?: boolean;
}

/**
 * Idempotently add `--json` to a commander Command. Commands opt in to
 * machine-readable output by calling this once during setup.
 */
export function withJsonSupport(cmd: Command): Command {
  const already = cmd.options.some((o) => o.long === "--json");
  if (already) return cmd;
  return cmd.option("--json", "Emit machine-readable JSON (no colours, no spinners)");
}

export function isJsonMode(opts: WithJsonOption): boolean {
  return opts.json === true;
}

/** Write a single JSON value to stdout with a trailing newline. */
export function writeJson(value: unknown): void {
  process.stdout.write(`${JSON.stringify(value)}\n`);
}

/** Write one JSON record per element to stdout (NDJSON). */
export function writeJsonStream(values: Iterable<unknown>): void {
  for (const v of values) {
    process.stdout.write(`${JSON.stringify(v)}\n`);
  }
}

/**
 * Write an error. In JSON mode, emits `{error,code}` to stderr. Otherwise,
 * a themed red line.
 */
export function writeError(opts: WithJsonOption, message: string, code?: string): void {
  if (isJsonMode(opts)) {
    const payload = code === undefined ? { error: message } : { error: message, code };
    process.stderr.write(`${JSON.stringify(payload)}\n`);
    return;
  }
  const meta = code === undefined ? undefined : code;
  process.stderr.write(`${theme.line(theme.glyph.cross, "err", message, meta)}\n`);
}
```

# Phase B — Migrate commands

Migration pattern for every command file:

1. Import `theme` and `withJsonSupport` / `writeJson` / `writeError`.
2. Replace every `console.log(...)` emitting user-facing text with an equivalent `theme.*` call routed through `process.stdout.write`.
3. Replace every `console.error(...)` with `writeError(opts, msg, code?)`.
4. Add `withJsonSupport(cmd)` to the top-level command builder.
5. Branch on `isJsonMode(opts)` at each output point: JSON mode emits one `writeJson` call per logical result; themed mode uses `theme.line` / `theme.heading` / `theme.kv` / `theme.table`.

Batched into two tasks to keep commits focused.

function toneForEventType(type: string): Parameters<typeof theme.line>[1] {
  if (type === "node_status") return "accent";
  if (type === "review_verdict") return "muted";
  if (type === "cost_update") return "dim";
  if (type === "graph_modified") return "muted";
  if (type === "agent_message") return "muted";
  if (type.endsWith("_error") || type === "workflow_status_failed") return "err";
  return "muted";
}

export function renderEvent(e: { type: string; timestamp: string; nodeId?: string }): string {
  const tone = toneForEventType(e.type);
  const glyph = tone === "err" ? theme.glyph.cross : theme.glyph.arrow;
  const meta = e.nodeId === undefined ? undefined : `node ${e.nodeId}`;
  return theme.line(glyph, tone, `${e.type}  ${theme.dim(e.timestamp)}`, meta);
}
